rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection - only admins can read/write all
    match /users/{userId} {
      allow read, write: if request.auth != null && (
        request.auth.uid == userId || // Users can read/write their own data
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN'
      );
    }
    
    // Staff collection - similar rules
    match /Staff/{staffId} {
      allow read, write: if request.auth != null && (
        request.auth.uid == staffId ||
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN'
      );
    }
    
    // Daily data - role-based access
    match /daily/{date} {
      allow read: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('daily');
      allow write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('daily');
      
      // Nested rows
      match /rows/{rowId} {
        allow read, write: if request.auth != null && 
          get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('daily');
      }
    }
    
    // Transport data
    match /transportData/{docId} {
      allow read, write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('transport-data');
    }
    
    // Prices
    match /prices/{docId} {
      allow read: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('prices');
      allow write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN';
    }
    
    // Drivers roles
    match /drivers/{docId} {
      allow read, write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('drivers-roles');
    }
    
    // Treasury system
    match /treasury/{docId} {
      allow read, write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('treasury-system');
    }
    
    // Garage
    match /garage/{docId} {
      allow read, write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('garage');
    }
    
    // Accounting
    match /accounting/{docId} {
      allow read, write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('accounting');
    }
    
    // Inventory
    match /Inventory/{itemId} {
      allow read: if request.auth != null; // All authenticated users can read
      allow write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('garage');
    }
    
    // Job Orders
    match /JobOrders/{jobId} {
      allow read: if request.auth != null; // All authenticated users can read
      allow write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.allowedPages.contains('garage');
    }
    
    // Change requests - only admins can approve
    match /changeRequests/{requestId} {
      allow read: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN';
      allow write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN';
    }
    
    // Shared lists - all authenticated users
    match /sharedLists/{listId} {
      allow read, write: if request.auth != null;
    }
  }
}
